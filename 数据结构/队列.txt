#include<stdlib.h>
#define maxsize 10
typedef struct queue
{
    int data[maxsize];
    int front;
    int rear;
}queue,*squeue;
void initqueue(squeue *q)
{
    *q=(squeue)malloc(sizeof(queue));
    (*q)->front=0;
    (*q)->rear=0;
}
void enqueue(squeue q,int data)
{
    if ((q->rear+1)%maxsize==q->front)
    {
        printf("Тњ");
        return ;
    }
    else
    {
        q->data[q->rear]=data;
        q->rear=(q->rear+1)%maxsize;
    }

}
void dequeue(squeue q)
{
    if (q->rear==q->front)
    {
        printf("Пе");
        return ;
    }
    else
    {
        printf("%d",q->data[q->front]);
        q->front=(q->front+1)%maxsize;
    }
}
int main ()
{
    squeue q;
    initqueue(&q);
    enqueue(q,1);
    enqueue(q,2);
    enqueue(q,3);
    dequeue(q);
    dequeue(q);
    dequeue(q);
}
