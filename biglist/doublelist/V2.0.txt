struct node
{
    int data;
    struct node * prior;
    struct node * next;
};
struct node * createlist()
{
    struct node * head=(struct node *)malloc(sizeof(struct node ));
    head->next=head;
    head->prior=head;
    return head;
};
struct node * createnode (int data)
{
    struct node *newnode =(struct node *)malloc( sizeof(struct node ));
    newnode->next=newnode->prior=NULL;
    newnode->data=data;
    return newnode;
};

void insertbytail(struct node *head,int data) //尾插法
{
    struct node *newnode=createnode(data);
    struct node *last=head;
    while (last->next!=head)
    {
        last=last->next;
    }
    newnode->prior=last;
    newnode->next=head;
    head->prior=newnode;
    last->next=newnode; //前驱后继->后继前驱->前去后继

int numdelete(struct node *head,int posdata)
{
    struct node *posnode=head->next;
    int num=0;
    while (posnode!=head)
    {
        if (posnode->data==posdata)
        {
            num++;
        }
        posnode=posnode->next;
    }
    return num;
}
void deletenode (struct node *head,int posdata)
{
    struct node * posnode=head->next;
    while (posnode->data!=posdata)
    {

        if (posnode==head)
        {
            printf("未找到该元素，无法删除");
            return ;
        }
        posnode=posnode->next;
    }
    posnode->prior->next=posnode->next;
    posnode->next->prior=posnode->prior;
    free(posnode);
}
void prinlist(struct node *head)
{
    struct node * move=head->next;
    while (move!=head)
    {
        printf("%4d",move->data);
        move=move->next;
    }
    printf("\n");
}

{
    struct node * move=head;
    struct node * newnode=createnode(newdata);
    int i;
    for (i=0; i<newpos; i++)
    {
        move=move->next;
        if (move==head)
        {
            printf("未找到该位置\n");
            return ;
        }
    }
    newnode->prior=move;
    newnode->next=move->next;
    move->next->prior=newnode;
    move->next=newnode;
}
void reverseprinlist (struct node *head)
{
    struct node *move=head->next;
    while (move->next!=head)
    {
        move=move->next;
    }
    while (move!=head)
    {
        printf("%4d",move->data);
        move=move->prior;
    }
    printf("\n");
}
